# Pinia Store Patterns

## Structure Template

```typescript
import { defineStore } from 'pinia'
import { ref, computed, type Ref, type ComputedRef } from 'vue'

interface StoreState {
  items: Item[]
  selectedId: string | null
  loading: boolean
}

interface StoreGetters {
  selectedItem: ComputedRef<Item | null>
  isEmpty: ComputedRef<boolean>
  itemCount: ComputedRef<number>
}

interface StoreActions {
  fetchItems: () => Promise<void>
  selectItem: (id: string) => void
  clearSelection: () => void
  reset: () => void
}

type UseItemStoreReturn = StoreState & StoreGetters & StoreActions

export const useItemStore = defineStore('item', (): UseItemStoreReturn => {
  // ═══════════════════════════════════════════════════════════
  // STATE
  // ═══════════════════════════════════════════════════════════
  const items = ref<Item[]>([])
  const selectedId = ref<string | null>(null)
  const loading = ref(false)

  // ═══════════════════════════════════════════════════════════
  // GETTERS (Computed)
  // ═══════════════════════════════════════════════════════════
  const selectedItem = computed((): Item | null => {
    if (!selectedId.value) return null
    return items.value.find((item) => item.id === selectedId.value) ?? null
  })

  const isEmpty = computed((): boolean => items.value.length === 0)

  const itemCount = computed((): number => items.value.length)

  // ═══════════════════════════════════════════════════════════
  // ACTIONS
  // ═══════════════════════════════════════════════════════════
  const fetchItems = async (): Promise<void> => {
    loading.value = true

    try {
      const response = await itemApi.getAll()
      items.value = response.data
    } catch (error) {
      console.error('Failed to fetch items:', error)
      throw error
    } finally {
      loading.value = false
    }
  }

  const selectItem = (id: string): void => {
    const exists = items.value.some((item) => item.id === id)
    if (!exists) {
      throw new Error(`Item with id ${id} not found`)
    }
    selectedId.value = id
  }

  const clearSelection = (): void => {
    selectedId.value = null
  }

  const reset = (): void => {
    items.value = []
    selectedId.value = null
    loading.value = false
  }

  // ═══════════════════════════════════════════════════════════
  // RETURN
  // ═══════════════════════════════════════════════════════════
  return {
    // State
    items,
    selectedId,
    loading,

    // Getters
    selectedItem,
    isEmpty,
    itemCount,

    // Actions
    fetchItems,
    selectItem,
    clearSelection,
    reset,
  }
})
```

## Naming Conventions

| Type       | Convention                | Example                               |
| ---------- | ------------------------- | ------------------------------------- |
| Store      | `use` + Domain + `Store`  | `useAuthStore`, `useUserStore`        |
| Store ID   | kebab-case domain         | `'auth'`, `'user-profile'`            |
| State vars | camelCase nouns           | `items`, `currentUser`, `isLoading`   |
| Getters    | camelCase, noun/adjective | `selectedItem`, `isEmpty`, `fullName` |
| Actions    | camelCase verbs           | `fetchItems`, `setUser`, `clearCache` |

## State Management Rules

### 1. Never Mutate State Directly from Components

```typescript
// ❌ BAD: Direct mutation in component
const store = useUserStore()
store.user.name = 'New Name' // Direct mutation!

// ✅ GOOD: Use action
const store = useUserStore()
store.updateUserName('New Name')
```

### 2. Use Actions for Side Effects

```typescript
// ❌ BAD: API call in component
const store = useUserStore()
const response = await api.updateUser(data)
store.user = response.data // Direct assignment!

// ✅ GOOD: Action handles everything
const store = useUserStore()
await store.updateUser(data) // API call + state update inside
```

### 3. Getters for Derived State

```typescript
// ❌ BAD: Computing in component
const store = useUserStore()
const fullName = computed(() => `${store.user?.firstName} ${store.user?.lastName}`)

// ✅ GOOD: Getter in store
// In store:
const fullName = computed((): string => {
  if (!user.value) return ''
  return `${user.value.firstName} ${user.value.lastName}`
})

// In component:
const store = useUserStore()
console.log(store.fullName)
```

## Common Patterns

### Auth Store

```typescript
export const useAuthStore = defineStore('auth', () => {
  // State
  const user = ref<User | null>(null)
  const token = ref<string | null>(null)
  const loading = ref(false)

  // Getters
  const isAuthenticated = computed((): boolean => !!token.value)
  const userRole = computed((): string | null => user.value?.role ?? null)

  // Actions
  const login = async (credentials: LoginCredentials): Promise<void> => {
    loading.value = true
    try {
      const response = await authApi.login(credentials)
      token.value = response.data.token
      await loadProfile()
    } finally {
      loading.value = false
    }
  }

  const logout = async (): Promise<void> => {
    try {
      await authApi.logout()
    } finally {
      // Always clear local state
      user.value = null
      token.value = null
    }
  }

  const loadProfile = async (): Promise<User | null> => {
    if (!token.value) return null

    try {
      const response = await authApi.getProfile()
      user.value = response.data
      return response.data
    } catch {
      // Token might be expired
      token.value = null
      return null
    }
  }

  return {
    user,
    token,
    loading,
    isAuthenticated,
    userRole,
    login,
    logout,
    loadProfile,
  }
})
```

### CRUD Store

```typescript
export const useResourceStore = defineStore('resource', () => {
  // State
  const items = ref<Resource[]>([])
  const currentItem = ref<Resource | null>(null)
  const loading = ref(false)
  const error = ref<Error | null>(null)

  // Getters
  const findById = computed(() => {
    return (id: string): Resource | undefined => {
      return items.value.find((item) => item.id === id)
    }
  })

  // Actions
  const fetchAll = async (): Promise<void> => {
    loading.value = true
    error.value = null
    try {
      const response = await resourceApi.getAll()
      items.value = response.data
    } catch (err) {
      error.value = err instanceof Error ? err : new Error('Fetch failed')
      throw error.value
    } finally {
      loading.value = false
    }
  }

  const fetchById = async (id: string): Promise<Resource> => {
    loading.value = true
    try {
      const response = await resourceApi.getById(id)
      currentItem.value = response.data
      return response.data
    } finally {
      loading.value = false
    }
  }

  const create = async (data: CreateResourceDto): Promise<Resource> => {
    const response = await resourceApi.create(data)
    items.value.push(response.data)
    return response.data
  }

  const update = async (id: string, data: UpdateResourceDto): Promise<Resource> => {
    const response = await resourceApi.update(id, data)
    const index = items.value.findIndex((item) => item.id === id)
    if (index !== -1) {
      items.value[index] = response.data
    }
    if (currentItem.value?.id === id) {
      currentItem.value = response.data
    }
    return response.data
  }

  const remove = async (id: string): Promise<void> => {
    await resourceApi.delete(id)
    items.value = items.value.filter((item) => item.id !== id)
    if (currentItem.value?.id === id) {
      currentItem.value = null
    }
  }

  return {
    items,
    currentItem,
    loading,
    error,
    findById,
    fetchAll,
    fetchById,
    create,
    update,
    remove,
  }
})
```

## Anti-Patterns

```typescript
// ❌ BAD: Options API style (harder to type, less flexible)
export const useStore = defineStore('store', {
  state: () => ({ count: 0 }),
  getters: { double: (state) => state.count * 2 },
  actions: {
    increment() {
      this.count++
    },
  },
})

// ✅ GOOD: Composition API style
export const useStore = defineStore('store', () => {
  const count = ref(0)
  const double = computed((): number => count.value * 2)
  const increment = (): void => {
    count.value++
  }
  return { count, double, increment }
})
```

```typescript
// ❌ BAD: Store does too much (>200 lines, multiple domains)
export const useAppStore = defineStore('app', () => {
  // Auth logic...
  // User profile logic...
  // Settings logic...
  // Notifications logic...
  // 500 lines later...
})

// ✅ GOOD: Split by domain
export const useAuthStore = defineStore('auth', () => {
  /* ... */
})
export const useUserStore = defineStore('user', () => {
  /* ... */
})
export const useSettingsStore = defineStore('settings', () => {
  /* ... */
})
export const useNotificationStore = defineStore('notification', () => {
  /* ... */
})
```

```typescript
// ❌ BAD: Exposing internal helpers
export const useStore = defineStore('store', () => {
  const _internalHelper = () => {
    /* ... */
  } // Should be private
  return { _internalHelper } // Don't expose!
})

// ✅ GOOD: Keep internals private
export const useStore = defineStore('store', () => {
  const internalHelper = (): void => {
    /* ... */
  }
  const publicAction = (): void => {
    internalHelper()
  }
  return { publicAction } // Only public API
})
```
