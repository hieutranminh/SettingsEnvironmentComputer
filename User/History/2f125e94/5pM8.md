# Ví dụ thực tế: Truy cập DOM của DataTable

## 1. Cách lấy DOM cơ bản

```typescript
// Trong component cha
const accessBasicDOM = (): void => {
  if (branchSalesTableRef.value) {
    // Lấy toàn bộ DOM của DataTable
    const tableDOM = branchSalesTableRef.value.getDataTableDOM()
    console.log('Full DataTable DOM:', tableDOM)

    // Lấy các phần cụ thể
    const wrapper = branchSalesTableRef.value.getDataTableWrapper()
    const header = branchSalesTableRef.value.getDataTableHeader()
    const body = branchSalesTableRef.value.getDataTableBody()
    const footer = branchSalesTableRef.value.getDataTableFooter()

    console.log('Wrapper:', wrapper)
    console.log('Header:', header)
    console.log('Body:', body)
    console.log('Footer:', footer)
  }
}
```

## 2. Lấy thông tin từ Rows và Columns

```typescript
const getTableInfo = (): void => {
  if (branchSalesTableRef.value) {
    const rows = branchSalesTableRef.value.getDataTableRows()
    const columns = branchSalesTableRef.value.getDataTableColumns()

    // Thông tin cơ bản
    console.log(`Table có ${rows?.length || 0} rows`)
    console.log(`Table có ${columns?.length || 0} columns`)

    // Lấy text content của từng row
    if (rows) {
      rows.forEach((row, index) => {
        const cells = row.querySelectorAll('td')
        const rowData = Array.from(cells).map((cell) => cell.textContent?.trim())
        console.log(`Row ${index}:`, rowData)
      })
    }

    // Lấy text content của từng column header
    if (columns) {
      columns.forEach((column, index) => {
        const headerText = column.textContent?.trim()
        console.log(`Column ${index}:`, headerText)
      })
    }
  }
}
```

## 3. Tùy chỉnh giao diện thông qua DOM

```typescript
const customizeTableAppearance = (): void => {
  if (branchSalesTableRef.value) {
    // Thêm class vào wrapper
    const wrapper = branchSalesTableRef.value.getDataTableWrapper()
    if (wrapper) {
      wrapper.classList.add('custom-wrapper')
      wrapper.style.border = '2px solid #007bff'
    }

    // Thay đổi style của header
    const header = branchSalesTableRef.value.getDataTableHeader()
    if (header) {
      header.style.backgroundColor = '#f8f9fa'
      header.style.fontWeight = 'bold'
    }

    // Thêm highlight cho rows
    const rows = branchSalesTableRef.value.getDataTableRows()
    if (rows) {
      rows.forEach((row, index) => {
        if (index % 2 === 0) {
          row.style.backgroundColor = '#f8f9fa'
        }
      })
    }
  }
}
```

## 4. Thêm Event Listeners

```typescript
const addEventListeners = (): void => {
  if (branchSalesTableRef.value) {
    const rows = branchSalesTableRef.value.getDataTableRows()

    if (rows) {
      rows.forEach((row, index) => {
        // Thêm click event
        row.addEventListener('click', (event) => {
          console.log(`Clicked row ${index}`)
          console.log('Event target:', event.target)

          // Highlight row được click
          rows.forEach((r) => r.classList.remove('selected-row'))
          row.classList.add('selected-row')
        })

        // Thêm hover effect
        row.addEventListener('mouseenter', () => {
          row.style.backgroundColor = '#e3f2fd'
        })

        row.addEventListener('mouseleave', () => {
          if (!row.classList.contains('selected-row')) {
            row.style.backgroundColor = ''
          }
        })
      })
    }
  }
}
```

## 6. Tạo báo cáo từ DOM

```typescript
const generateReportFromDOM = (): void => {
  if (branchSalesTableRef.value) {
    const rows = branchSalesTableRef.value.getDataTableRows()

    if (rows) {
      let totalServiceSales = 0
      let totalProductSales = 0
      let totalSales = 0
      const branchData = []

      rows.forEach((row) => {
        const cells = row.querySelectorAll('td')
        const cellArray = Array.from(cells)

        const serviceSales = parseFloat(cellArray[1]?.textContent?.replace(/[^\d.-]/g, '') || '0')
        const productSales = parseFloat(cellArray[4]?.textContent?.replace(/[^\d.-]/g, '') || '0')
        const salesTotal = parseFloat(cellArray[11]?.textContent?.replace(/[^\d.-]/g, '') || '0')

        totalServiceSales += serviceSales
        totalProductSales += productSales
        totalSales += salesTotal

        branchData.push({
          branch: cellArray[0]?.textContent?.trim(),
          serviceSales,
          productSales,
          salesTotal,
        })
      })

      const report = {
        summary: {
          totalBranches: rows.length,
          totalServiceSales,
          totalProductSales,
          totalSales,
        },
        branchDetails: branchData,
      }

      console.log('Generated report:', report)
      return report
    }
  }
}
```

## 7. Export DOM thành HTML

```typescript
const exportTableAsHTML = (): string => {
  if (branchSalesTableRef.value) {
    const tableDOM = branchSalesTableRef.value.getDataTableDOM()

    if (tableDOM) {
      // Clone DOM để không ảnh hưởng đến original
      const clonedTable = tableDOM.cloneNode(true) as HTMLElement

      // Thêm CSS styles
      const styles = `
        <style>
          .p-datatable { border-collapse: collapse; width: 100%; }
          .p-datatable th, .p-datatable td { border: 1px solid #ddd; padding: 8px; text-align: left; }
          .p-datatable th { background-color: #f2f2f2; }
          .p-datatable tr:nth-child(even) { background-color: #f9f9f9; }
        </style>
      `

      const htmlContent = `
        <!DOCTYPE html>
        <html>
        <head>
          <title>Branch Sales Report</title>
          ${styles}
        </head>
        <body>
          <h1>Branch Sales Report</h1>
          ${clonedTable.outerHTML}
        </body>
        </html>
      `

      return htmlContent
    }
  }

  return ''
}
```

## 8. Sử dụng với Print functionality

```typescript
const printTable = (): void => {
  if (branchSalesTableRef.value) {
    const tableDOM = branchSalesTableRef.value.getDataTableDOM()

    if (tableDOM) {
      // Tạo window mới để print
      const printWindow = window.open('', '_blank')
      if (printWindow) {
        const htmlContent = `
          <!DOCTYPE html>
          <html>
          <head>
            <title>Branch Sales Report</title>
            <style>
              body { font-family: Arial, sans-serif; }
              .p-datatable { border-collapse: collapse; width: 100%; }
              .p-datatable th, .p-datatable td { border: 1px solid #ddd; padding: 8px; }
              .p-datatable th { background-color: #f2f2f2; }
              @media print {
                body { margin: 0; }
                .p-datatable { page-break-inside: auto; }
              }
            </style>
          </head>
          <body>
            <h1>Branch Sales Report</h1>
            ${tableDOM.outerHTML}
          </body>
          </html>
        `

        printWindow.document.write(htmlContent)
        printWindow.document.close()
        printWindow.print()
      }
    }
  }
}
```

## Lưu ý quan trọng

1. **Performance**: Truy cập DOM thường xuyên có thể ảnh hưởng đến performance
2. **Reactivity**: DOM có thể thay đổi khi data thay đổi, cần cập nhật lại
3. **Type Safety**: Luôn kiểm tra null/undefined trước khi sử dụng
4. **Memory Leaks**: Nhớ remove event listeners khi component unmount
5. **Browser Compatibility**: Một số DOM methods có thể không hoạt động trên tất cả browsers
